import { Component, Input, Output, EventEmitter } from '@angular/core';
import { FormGroup } from '@angular/forms';

@Component({
  selector: 'app-card-input',
  templateUrl: './card-input.component.html',
  styleUrls: ['./card-input.component.scss'],
})
export class CardInputComponent {
  focusOn() {
    this.focusOnEmit.emit();
  }
  focusOut() {
    this.focusOutEmit.emit();
  }

  @Input() submitHistoryFlag: Boolean;
  @Input() formGroup: FormGroup;
  @Input() errors: string;
  @Input() controlName: string;
  @Input() label: string;
  @Input() placeholder: string;
  @Output() focusOnEmit = new EventEmitter();
  @Output() focusOutEmit = new EventEmitter();
}

<div [formGroup]="formGroup">
  <label
    for="{{ controlName }}"
    class="block text-sm font-medium text-gray-700 mb-2"
    [ngClass]="[
      !submitHistoryFlag
        ? 'text-black-700 dark:text-black-500'
        : formGroup.controls[controlName].errors !== null
        ? 'text-red-700 dark:text-red-500'
        : 'text-green-700 dark:text-green-500'
    ]"
    >{{ label }}</label
  >
  <input
    type="text"
    [name]="controlName"
    [id]="controlName"
    [placeholder]="placeholder"
    class="w-full py-3 px-4 border border-gray-400 rounded-lg focus:outline-none focus:border-blue-500"
    [formControlName]="controlName"
    autocomplete="off"
    [ngClass]="[
      !submitHistoryFlag
        ? 'bg-black-50 border-black-500 text-black-900 placeholder-black-700 focus:ring-black-500 focus:border-black-500 dark:bg-black-100 dark:border-black-400'
        : formGroup.controls[controlName].errors !== null
        ? 'bg-red-50 border-red-500 text-red-900 placeholder-red-700 focus:ring-red-500 focus:border-red-500 dark:bg-red-100 dark:border-red-400'
        : 'bg-green-50 border-green-500 text-green-900 placeholder-green-700 focus:ring-green-500 focus:border-green-500 dark:bg-green-100 dark:border-green-400'
    ]"
    (focus)="focusOn()"
    (focusout)="focusOut()"
  />
</div>
